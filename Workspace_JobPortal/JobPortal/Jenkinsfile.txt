pipeline {
   agent any

   stages {
       stage('Checkout') {
           steps {
               git 'https://github.com/gaurav-lucky/j-portal'
           }
       }

       stage('Build') {
           steps {
               sh 'mvn clean install'
           }
       }

       stage('Test') {
           steps {
               sh 'mvn test'
           }
       }

       stage('Package') {
           steps {
               sh 'mvn package'
           }
       }
   }
}
